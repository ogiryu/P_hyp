積立NISAシミュレーション




import matplotlib.pyplot as plt from pyscript import Element def run_simulation(event): amount = int(Element("amount").value) rate = float(Element("rate").value) / 100 / 12 years = int(Element("years").value) total = 0 xs = [] ys_total = [] ys_principal = [] for i in range(years * 12): total = total * (1 + rate) total += amount if (i + 1) % 12 == 0: xs.append((i + 1) // 12) ys_total.append(total / 10000) ys_principal.append(amount * (i + 1) / 10000) # 結果表示 Element("output").write( f"元本：{amount * years * 12} 円
" f"最終金額：{int(total)} 円
" f"利益：{int(total - amount * years * 12)} 円" ) # グラフ描画 fig, ax = plt.subplots() ax.plot(xs, ys_total, label="total") ax.plot(xs, ys_principal, label="principal") ax.set_title("NISA") ax.set_xlabel("year") ax.set_ylabel("amount [ten thousand yen]") ax.grid(True) ax.legend() Element("plot").clear() Element("plot").write(fig) Element("run-btn").element.onclick = run_simulation
